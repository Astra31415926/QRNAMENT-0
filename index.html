/** * СИСТЕМА ГЕНЕРАЦИИ СИММЕТРИЧНОГО ОРНАМЕНТА (D8) * Синхронизация: 1 пиксель орнамента = 1 модуль QR */ const setupUI = () => { const container = document.createElement('div'); container.style.cssText = 'display:flex; flex-direction:column; align-items:center; gap:20px; padding:20px; font-family:sans-serif;'; const input = document.createElement('input'); input.type = 'text'; input.placeholder = 'Введите текст для генерации...'; input.style.cssText = 'width:300px; padding:10px; border-radius:5px; border:1px solid #ccc;'; const canvas = document.createElement('canvas'); canvas.style.imageRendering = 'pixelated'; // Критично для сохранения четкости модулей document.body.appendChild(container); container.appendChild(input); container.appendChild(canvas); return { input, canvas, ctx: canvas.getContext('2d') }; }; const { input, canvas, ctx } = setupUI(); function generate(text) { // 1. ОПРЕДЕЛЕНИЕ РАЗМЕРА (Синхронизация с QR) // Версия 1 = 21x21, Версия 2 = 25x25 и т.д. const qrGridDim = 21 + (Math.floor(text.length / 3) * 4); const moduleSize = 12; // Размер одного физического пикселя модуля const totalSize = qrGridDim * moduleSize; canvas.width = canvas.height = totalSize; const center = Math.floor(qrGridDim / 2); // --- ФАЗА 1: ДИСКРЕТНАЯ ГЕНЕРАЦИЯ (ЛОГИКА) --- // Работаем только с 1/8 частью (октант) const masterOctant = []; const seed = text.split('').reduce((acc, char) => acc + char.charCodeAt(0), 0); for (let y = 0; y <= center; y++) { masterOctant[y] = []; for (let x = 0; x <= y; x++) { // Математическое "зерно" на основе текста const hash = Math.abs(Math.sin(seed + x * 0.7 + y * 0.3) * 10000); masterOctant[y][x] = { active: hash % 2 > 0.8, // 0 или 1 color: `hsl(${(seed + hash) % 360}, 60%, 50%)` }; } } // --- ФАЗА 2: РЕНДЕР (ТОЧКА НЕВОЗВРАТА) --- ctx.clearRect(0, 0, totalSize, totalSize); // Отрисовка всех 8 отражений на основе мастер-октанта for (let y = 0; y <= center; y++) { for (let x = 0; x <= y; x++) { const cell = masterOctant[y][x]; if (cell.active) { ctx.fillStyle = cell.color; drawD8(x, y, center, moduleSize); } } } // --- ФАЗА 3: УСИЛЕНИЕ СИММЕТРИИ (ТОЛЬКО ПИКСЕЛИ) --- applyRasterFilters(totalSize); } // Функция отрисовки 8 зеркальных точек (строго по сетке модулей) function drawD8(x, y, center, size) { const symPoints = [ [center + x, center + y], [center - x, center + y], [center + x, center - y], [center - x, center - y], [center + y, center + x], [center - y, center + x], [center + y, center - x], [center - y, center - x] ]; symPoints.forEach(([px, py]) => { ctx.fillRect(px * size, py * size, size, size); }); } // Работа с растром после уничтожения структуры блоков function applyRasterFilters(totalSize) { const tempCanvas = document.createElement('canvas'); tempCanvas.width = tempCanvas.height = totalSize; const tCtx = tempCanvas.getContext('2d'); // Копируем текущий кадр, вращаем на 90 градусов tCtx.translate(totalSize / 2, totalSize / 2); tCtx.rotate(Math.PI / 2); tCtx.drawImage(canvas, -totalSize / 2, -totalSize / 2); // Накладываем через разность (эффект интерференции) ctx.globalCompositeOperation = 'difference'; ctx.drawImage(tempCanvas, 0, 0); ctx.globalCompositeOperation = 'source-over'; } input.addEventListener('input', (e) => generate(e.target.value || " ")); // Инициализация generate("MANDALA_QR");
