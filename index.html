function drawSnowflake(ctx, nCount, pxSize, usePats, pal, smart) { const imgSize = nCount * pxSize; const gridDim = Math.floor(nCount / CELL) + 5; const center = Math.floor(gridDim / 2); const quad = center + 1; const blockPx = CELL * pxSize; // Размер одного блока (7x7 пикселей) // 1. Генерируем данные только для одного треугольника (октанта) let octant = []; for (let r = 0; r < quad; r++) { octant[r] = []; for (let c = 0; c <= r; c++) { let color = pal[Math.floor(Math.random() * pal.length)]; let pat = usePats ? PATS[Object.keys(PATS)[Math.floor(Math.random() * Object.keys(PATS).length)]] : PATS['s']; if (smart && !usePats) { pat = PATS[FILL_KEYS[Math.floor(Math.random() * FILL_KEYS.length)]]; color = color > 128 ? 20 : 230; } octant[r][c] = { pat, color }; } } // 2. СОЗДАЕМ ВРЕМЕННЫЙ ХОЛСТ ДЛЯ СЕГМЕНТА (ЗАПЕКАНИЕ) const tempCanvas = document.createElement('canvas'); tempCanvas.width = imgSize; tempCanvas.height = imgSize; const tCtx = tempCanvas.getContext('2d'); const offset = (imgSize - gridDim * blockPx) / 2; // Рисуем один сегмент (треугольник) for (let r = 0; r < quad; r++) { for (let c = 0; c <= r; c++) { const data = octant[r][c]; const startX = offset + (center + c) * blockPx; const startY = offset + (center + r) * blockPx; tCtx.fillStyle = `rgb(${data.color},${data.color},${data.color})`; const pSize = blockPx / 7; for (let i = 0; i < 49; i++) { if (data.pat[i]) { const px = i % 7; const py = Math.floor(i / 7); tCtx.fillRect(startX + px * pSize, startY + py * pSize, pSize + 0.5, pSize + 0.5); } } } } // 3. ОТРАЖАЕМ ЗАПЕЧЕННЫЙ СЕГМЕНТ 8 РАЗ ctx.save(); ctx.translate(imgSize / 2, imgSize / 2); // Сдвиг в центр for (let i = 0; i < 8; i++) { ctx.save(); ctx.rotate((i * 45) * Math.PI / 180); if (i % 2 === 1) ctx.scale(1, -1); // Рисуем сегмент так, чтобы его угол был в центре ctx.drawImage(tempCanvas, -imgSize / 2, -imgSize / 2); ctx.restore(); } ctx.restore(); }
