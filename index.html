/** * ГЕНЕРАТОР "ЦИФРОВОЙ КРИСТАЛЛ" (QR-СИНХРОНИЗАЦИЯ) */ const setupUI = () => { document.body.style.backgroundColor = '#1a1a1a'; const container = document.createElement('div'); container.style.cssText = 'display:flex; flex-direction:column; align-items:center; gap:20px; padding:40px;'; const input = document.createElement('input'); input.type = 'text'; input.placeholder = 'Введите текст...'; input.style.cssText = 'width:400px; padding:12px; background:#333; color:#fff; border:1px solid #555; outline:none;'; const canvas = document.createElement('canvas'); canvas.style.boxShadow = '0 0 50px rgba(0,0,0,0.5)'; canvas.style.imageRendering = 'pixelated'; document.body.appendChild(container); container.appendChild(input); container.appendChild(canvas); return { input, canvas, ctx: canvas.getContext('2d') }; }; const { input, canvas, ctx } = setupUI(); function generate(text) { // 1. СЕТКА (Связь с версией QR: 21, 25, 29...) const qrGridDim = 21 + (Math.floor(text.length / 2) * 4); const moduleSize = Math.floor(600 / qrGridDim); // Адаптивный размер модуля const totalSize = qrGridDim * moduleSize; canvas.width = canvas.height = totalSize; const center = Math.floor(qrGridDim / 2); const seed = text.split('').reduce((acc, char, i) => acc + char.charCodeAt(0) * (i + 1), 0); // --- ФАЗА 1: ГЕНЕРАЦИЯ ДАННЫХ --- const octant = []; for (let y = 0; y <= center; y++) { octant[y] = []; for (let x = 0; x <= y; x++) { const noise = Math.abs(Math.sin(seed + x * 0.5 + y * 0.8) * 10000); octant[y][x] = { active: (noise % 10) > 4, hue: (seed + noise) % 360 }; } } // --- ФАЗА 2: РЕНДЕР (БАЗОВЫЙ СЛОЙ) --- ctx.fillStyle = '#000'; ctx.fillRect(0, 0, totalSize, totalSize); for (let y = 0; y <= center; y++) { for (let x = 0; x <= y; x++) { if (octant[y][x].active) { ctx.fillStyle = `hsl(${octant[y][x].hue}, 80%, 50%)`; drawD8(x, y, center, moduleSize); } } } // --- ФАЗА 3: ПИКСЕЛЬНОЕ УСИЛЕНИЕ (ТОЛЬКО РАСТР) --- applyRasterSymmetry(totalSize); } function drawD8(x, y, center, size) { const points = [ [center+x, center+y], [center-x, center+y], [center+x, center-y], [center-x, center-y], [center+y, center+x], [center-y, center+x], [center+y, center-x], [center-y, center-x] ]; points.forEach(([px, py]) => ctx.fillRect(px * size, py * size, size, size)); } function applyRasterSymmetry(totalSize) { const temp = document.createElement('canvas'); temp.width = temp.height = totalSize; const tCtx = temp.getContext('2d'); // Делаем копию и поворачиваем на 90 градусов tCtx.translate(totalSize / 2, totalSize / 2); tCtx.rotate(Math.PI / 2); tCtx.drawImage(canvas, -totalSize / 2, -totalSize / 2); // Смешивание слоев (Difference создает неоновые контуры на стыках) ctx.globalCompositeOperation = 'difference'; ctx.drawImage(temp, 0, 0); // Добавляем финальный проход для "глубины" (Screen/Lighten) ctx.globalCompositeOperation = 'screen'; ctx.filter = 'contrast(150%) brightness(120%)'; ctx.drawImage(canvas, 0, 0); } input.addEventListener('input', (e) => generate(e.target.value || "CRYSTAL")); generate("MANDALA");
