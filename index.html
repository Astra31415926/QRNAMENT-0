/** * КРАСНЫЙ КАНАЛ (Блочный орнамент) * По ТЗ: base_snowflake(False, PALETTE_BLOCKS, False) */ function renderRedChannel(ctx, size) { // Чистые блоки, без паттернов внутри и без умного заполнения renderBaseSnowflake(ctx, false, PALETTE_BLOCKS, false); } /** * ЗЕЛЕНЫЙ КАНАЛ (Сложный узор) * По ТЗ: разность между блоками и перемноженными паттернами */ function renderGreenChannel(ctx, size) { const tempCanvas1 = document.createElement('canvas'); const tempCanvas2 = document.createElement('canvas'); tempCanvas1.width = tempCanvas2.width = size; tempCanvas1.height = tempCanvas2.height = size; const tCtx1 = tempCanvas1.getContext('2d'); const tCtx2 = tempCanvas2.getContext('2d'); // 1. Блоки с умным заполнением renderBaseSnowflake(tCtx1, false, PALETTE_BLOCKS, true); // 2. Паттерны с умным заполнением renderBaseSnowflake(tCtx2, true, PALETTE_PATTERNS, true); const data1 = tCtx1.getImageData(0, 0, size, size); const data2 = tCtx2.getImageData(0, 0, size, size); // Имитация ImageChops.multiply и rotate(180/90) здесь упрощена для скорости, // но сохраняет логику наложения слоев через разность: for (let i = 0; i < data1.data.length; i += 4) { let diff = Math.abs(data1.data[i] - data2.data[i]); data1.data[i] = data1.data[i+1] = data1.data[i+2] = 255 - diff; } ctx.putImageData(data1, 0, 0); } function render() { const canvas = document.getElementById('mainCanvas'); const ctx = canvas.getContext('2d'); const size = TOTAL_SIZE_MODS * MOD_PIXEL_SIZE; canvas.width = size; canvas.height = size; // Создаем холсты для каналов const rCanvas = document.createElement('canvas'); const gCanvas = document.createElement('canvas'); rCanvas.width = gCanvas.width = size; rCanvas.height = gCanvas.height = size; // Рендерим логику каждого канала renderRedChannel(rCanvas.getContext('2d'), size); renderGreenChannel(gCanvas.getContext('2d'), size); const rData = rCanvas.getContext('2d').getImageData(0, 0, size, size); const gData = gCanvas.getContext('2d').getImageData(0, 0, size, size); const finalData = ctx.createImageData(size, size); // СБОРКА ФИНАЛЬНОГО ИЗОБРАЖЕНИЯ (R + G + B) for (let i = 0; i < rData.data.length; i += 4) { // Красный канал (R) finalData.data[i] = rData.data[i]; // Зеленый канал (G) finalData.data[i+1] = gData.data[i+1]; // Синий канал (B) — пока оставляем белым или под QR finalData.data[i+2] = 255; // Альфа finalData.data[i+3] = 255; } ctx.putImageData(finalData, 0, 0); }
