const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d'); document.body.appendChild(canvas); const input = document.createElement('input'); input.type = 'text'; input.placeholder = 'Введите текст для генерации...'; document.body.appendChild(input); /** * КРИТИЧЕСКАЯ СИНХРОНИЗАЦИЯ: * 1 пиксель орнамента = 1 модуль QR-кода. */ function generateMandala(text) { // Имитация получения данных QR (размер N x N) // В реальном приложении здесь библиотека qrcode-generator const qrSize = Math.max(21, Math.min(57, text.length + 21)); const moduleSize = 10; // Размер одного модуля в экранных пикселях const dim = qrSize; canvas.width = canvas.height = dim * moduleSize; // ФАЗА 1: ДИСКРЕТНАЯ ГЕНЕРАЦИЯ (Логика в памяти) const grid = Array(dim).fill().map(() => Array(dim).fill(0)); const seed = text.split('').reduce((acc, char) => acc + char.charCodeAt(0), 0); // Заполнение мастер-октантa (1/8 часть) const center = Math.floor(dim / 2); for (let y = 0; y <= center; y++) { for (let x = 0; x <= y; x++) { // Псевдорандом на основе текста const val = (Math.sin(seed + x * 0.5 + y * 0.8) > 0.3) ? 1 : 0; grid[y][x] = val; } } // ФАЗА 2: РЕНДЕР (ТОЧКА НЕВОЗВРАТА) // Отражение октанта для формирования полной сетки ctx.fillStyle = "black"; ctx.fillRect(0, 0, canvas.width, canvas.height); for (let y = 0; y <= center; y++) { for (let x = 0; x <= y; x++) { if (grid[y][x]) { drawSymmetricPixels(x, y, center, moduleSize); } } } // ФАЗА 3: ПОСТ-ОБРАБОТКА (ТОЛЬКО ПИКСЕЛИ) applyPixelSymmetry(dim, moduleSize); } function drawSymmetricPixels(x, y, center, size) { const points = [ [center + x, center + y], [center - x, center + y], [center + x, center - y], [center - x, center - y], [center + y, center + x], [center - y, center + x], [center + y, center - x], [center - y, center - x] ]; ctx.fillStyle = `hsl(${(x * 10) % 360}, 70%, 50%)`; points.forEach(([px, py]) => { ctx.fillRect(px * size, py * size, size, size); }); } function applyPixelSymmetry(dim, size) { const tempCanvas = document.createElement('canvas'); tempCanvas.width = tempCanvas.height = canvas.width; const tCtx = tempCanvas.getContext('2d'); // Операция rotate(180) и наложение (Multiply/Difference) tCtx.translate(canvas.width / 2, canvas.height / 2); tCtx.rotate(Math.PI); tCtx.drawImage(canvas, -canvas.width / 2, -canvas.height / 2); ctx.globalCompositeOperation = 'difference'; ctx.drawImage(tempCanvas, 0, 0); } input.addEventListener('input', (e) => generateMandala(e.target.value)); generateMandala("START");
